425 /**
426  * Accepts plain text input and Gloriously WTFifies it.
427  * @author <a href="mailto:matthewkastor@gmail.com">
428  * Matthew Christopher Kastor-Inare III </a><br />
429  * ☭ Hial Atropa!! ☭
430  * @version 20120909
431  * @param {String} target The text to WTFify.
432  * @return {String} Returns Genuine WTFified text.
433  */
434 atropa.wtf.wtfify = function (target) {
435 	"use strict";
436 	var regexValue,
437 	replacementText,
438 	x,
439 	oldWord,
440 	wtfCount,
441 	wordCount,
442 	ret;
443 	
444 	ret = {};
445 	wtfCount = 0;
446 	target = target.trim();
447 	wordCount = atropa.string.countWords(target);
448 	/**
449 	 * Accepts plain text input and Gloriously WTFifies it.
450 	 * @author <a href="mailto:matthewkastor@gmail.com">
451 	 * Matthew Christopher Kastor-Inare III </a><br />
452 	 * ☭ Hial Atropa!! ☭
453 	 * @version 20120909
454 	 * @methodOf atropa.wtf.wtfify-
455 	 * @private
456 	 * @param {String} m First matched pattern in string searched.
457 	 * @param {String} sub1 First matched subpattern in string searched.
458 	 * @param {String} sub2 Second matched subpattern in string searched.
459 	 */
460 	replacementText = function (m, sub1, sub2) {
461 		wtfCount++;
462 		return sub1 + atropa.wtf.dictionary[x] + sub2;
463 	};
464 	for (x in atropa.wtf.dictionary) {
465 		if (atropa.wtf.dictionary.hasOwnProperty(x)) {
466 			oldWord = atropa.regex.appendPrefixesAndSuffixes(x);
467 			regexValue = new RegExp(oldWord, 'gi');
468 			target = target.replace(regexValue, replacementText);
469 		}
470 	}
471 	ret.wtfCount = wtfCount;
472 	ret.wordCount = wordCount;
473 	ret.score = wtfCount / wordCount;
474 	ret.txt = target;
475 	return ret;
476 };
