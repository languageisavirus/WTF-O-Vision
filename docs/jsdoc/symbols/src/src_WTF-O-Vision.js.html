 34 /**
 35  * Checks whether this class has been marked as unsupported and throws an 
 36  *  error if it has.
 37  * @author <a href="mailto:matthewkastor@gmail.com">
 38  *  Matthew Christopher Kastor-Inare III </a><br />
 39  *  ☭ Hial Atropa!! ☭
 40  * @version 20130308
 41  * @param {String} className The name of the class.
 42  * @param {String} errorMessage Optional. A custom error message. Defaults to
 43  *  atropa.data[className].error
 44  */
 45 atropa.supportCheck = function (className, errorMessage) {
 46     "use strict";
 47     className = String(className);
 48     errorMessage = errorMessage || atropa.data[className].error;
 49     errorMessage = String(errorMessage);
 50     
 51     if(atropa.data[className].support === 'unsupported') {
 52         throw new Error(errorMessage);
 53     }
 54 };
 55 /**
 56  * Pushes a requirement check into atropa.data.requirements. The test
 57  *  tests whether the class is supported in this environment. Sets
 58  *  atropa.data[className]'s support to unsupported and error to errorMessage
 59  *  if the requirementFn returns false. The requirement checks will all be run
 60  *  after the library has loaded.
 61  * @author <a href="mailto:matthewkastor@gmail.com">
 62  *  Matthew Christopher Kastor-Inare III </a><br />
 63  *  ☭ Hial Atropa!! ☭
 64  * @version 20130308
 65  * @param {String} className The name of the class.
 66  * @param {Function} requirementFn A function to test whether or not the class
 67  *  is supported in this environment. If supported, returns true otherwise
 68  *  return false.
 69  * @param {String} errorMessage The error message to use when this class or its
 70  *  methods are called in unsupported environments. Defaults to:
 71  *  'The atropa.' + className + ' class is unsupported in this environment.';
 72  */
 73 atropa.requires = function (className, requirementFn, errorMessage) {
 74     "use strict";
 75     var check = function () {
 76         var test = false;
 77         if(typeof className !== 'string') {
 78             throw new Error('atropa.requires requires the class name to be ' +
 79                 'specified');
 80         }
 81         
 82         if(atropa.data[className] === undefined) {
 83             atropa.data[className] = {};
 84             
 85             if(typeof requirementFn !== 'function') {
 86                 requirementFn = function () { return false; };
 87             }
 88             
 89             if(typeof errorMessage !== 'string') {
 90                 errorMessage = 'The atropa.' + className +
 91                     ' class is unsupported in this environment.';
 92             }
 93             
 94             try {
 95                 test = requirementFn();
 96             } catch (e) {
 97                 test = false;
 98             }
 99             
100             atropa.data[className].error = errorMessage;
101             
102             if(test === false) {
103                 atropa.data[className].support = 'unsupported';
104             }
105         }
106     };
107     
108     atropa.data.requirements.push(check);
109 };

165 /**
166  * Compares two arrays based on size, contents, and element order.
167  * @author <a href="mailto:matthewkastor@gmail.com">
168  *  Matthew Christopher Kastor-Inare III </a><br />
169  *  ☭ Hial Atropa!! ☭
170  * @version 20120909
171  * @param {Array} array1 One array you want compared to another.
172  * @param {Array} array2 The other array.
173  * @returns {Boolean} Returns true or false depending on
174  *  whether or not the arrays matched in size, composition, and
175  *  element order.
176  * @example
177  * var x = [1,2];
178  * var y = [1,1,3];
179  * atropa.arrays.match(x,y);
180  * // returns false
181  * @example
182  * var x = [1,2];
183  * var y = [1,2];
184  * atropa.arrays.match(x,y);
185  * // returns true
186  * @example
187  * var x = [1,2];
188  * var y = [2,1];
189  * atropa.arrays.match(x,y);
190  * // returns false because the elements are not in the same order.
191  * @example
192  * var x = [1,{'aProp' : 'aValue'}];
193  * var y = [1,{'aProp' : 'aValue'}];
194  * atropa.arrays.match(x,y);
195  * // returns false because even though the object looks the same, the
196  * // two objects are in fact distinct objects.
197  * @example
198  * var obj = {'aProp' : 'aValue'};
199  * var x = [1,obj];
200  * var y = [1,obj];
201  * atropa.arrays.match(x,y);
202  * // returns true because the objects referenced in the arrays are
203  * // in fact the same object.
204  */
205 atropa.arrays.match = function arraysMatch(array1, array2) {
206     "use strict";
207     var x,
208     l;
209     if (array1.length !== array2.length) {
210         return false;
211     }
212     l = array1.length;
213     for (x = 0; x < l; x += 1) {
214         if (array1[x] !== array2[x]) {
215             return false;
216         }
217     }
218     return true;
219 };
220 /**
221  * Subtracts one array from another array based on the unique values in both
222  *  sets.
223  * @author <a href="mailto:matthewkastor@gmail.com">
224  *  Matthew Christopher Kastor-Inare III </a><br />
225  *  ☭ Hial Atropa!! ☭
226  * @version 20130112
227  * @param {Array} a (subtrahend) The array to subtract.
228  * @param {Array} (minuend) fromB The array with elements duplicated in <code>a</code>
229  * @returns {Array} Returns a new array containing only the unique
230  *  values found in <code>fromB</code> that are not present in <code>a</code>
231  * @example
232  * var x = [1,2];
233  * var y = [1,1,3];
234  * atropa.arrays.subtract(x,y);
235  * // returns [3]
236  * @example
237  * var x = [1,3];
238  * var y = [3,1];
239  * atropa.arrays.subtract(x,y);
240  * // returns []
241  * @example
242  * var x = [1,3];
243  * var y = [3,1,1,9];
244  * atropa.arrays.subtract(x,y);
245  * // returns [9]
246  * @example
247  * var x = [1,3,{'aProp' : 'aVal'}];
248  * var y = [3,1,{'aProp' : 'aVal'}];
249  * atropa.arrays.subtract(x,y);
250  * // returns [{'aProp' : 'aVal'}] 
251  * // because the two objects are not the same object.
252  * @example
253  * var obj = {'aProp' : 'aVal'};
254  * var x = [1,3,obj];
255  * var y = [3,1,{'aProp' : 'aVal'}];
256  * atropa.arrays.subtract(x,y);
257  * // returns [{'aProp' : 'aVal'}] 
258  * // because the two objects are not the same object.
259  * @example
260  * var obj = {'aProp' : 'aVal'}
261  * var x = [1,3,obj];
262  * var y = [3,1,obj];
263  * atropa.arrays.subtract(x,y);
264  * // returns [] 
265  * // because the objects referenced in the arrays are the same object.
266  */
267 atropa.arrays.subtract = function(a, fromB) {
268     "use strict";
269     var the = {};
270     the.result = [];
271     fromB.forEach(function(item){
272         the.mark = false;
273         a.forEach(function(rm){
274             if(item === rm) {
275                 the.mark = true;
276             }
277         });
278         if(the.mark !== true) {
279             the.result.push(item);
280         }
281     });
282     return the.result;
283 };
284 /**
285  * Returns an array of values found in both of the given arrays.
286  * @author <a href="mailto:matthewkastor@gmail.com">
287  *  Matthew Christopher Kastor-Inare III </a><br />
288  *  ☭ Hial Atropa!! ☭
289  * @version 20130112
290  * @param {Array} array1 An array.
291  * @param {Array} array2 Another array.
292  * @returns {Array} Returns an array of values found in both of the given
293  *  arrays.
294  * @example
295  * var x = [1,3,4];
296  * var y = [3,1,5];
297  * atropa.arrays.intersect(x,y);
298  * // returns [1,3]
299  * @example
300  * var x = [1,1,3,4];
301  * var y = [3,1,1,5];
302  * atropa.arrays.intersect(x,y);
303  * // returns [1,1,3]
304  * @example
305  * var obj = {'aProp' : 'aVal'};
306  * var x = [1,3,obj];
307  * var y = [3,1,obj];
308  * atropa.arrays.intersect(x,y);
309  * // returns [1,3,{'aProp' : 'aVal'}]
310  * @example
311  * var obj = {'aProp' : 'aVal'};
312  * var x = [1,3,{'aProp' : 'aVal'}];
313  * var y = [3,1,obj];
314  * atropa.arrays.intersect(x,y);
315  * // returns [1,3] because the two objects are not the same object.
316  * @example
317  * var x = [1,3,{'aProp' : 'aVal'}];
318  * var y = [3,1,{'aProp' : 'aVal'}];
319  * atropa.arrays.intersect(x,y);
320  * // returns [1,3] because the two objects are not the same object.
321  */
322 atropa.arrays.intersect = function intersect(array1, array2) {
323     "use strict";
324     var smallArray, largeArray, intersection = [];
325     if(array1.length > array2.length) {
326         largeArray = array1.splice(0);
327         smallArray = array2.splice(0);
328     } else {
329         largeArray = array2.splice(0);
330         smallArray = array1.splice(0);
331     }
332     smallArray.forEach(function (item) {
333         var idxInLargeArray = largeArray.indexOf(item);
334         if (0 <= idxInLargeArray) { // has word
335             intersection.push(largeArray.splice(idxInLargeArray, 1)[0]);
336         }
337     });
338     return intersection;
339 };
340 /**
341  * Calculates the frequency of items occurring in an array.
342  * @author <a href="mailto:matthewkastor@gmail.com">
343  *  Matthew Christopher Kastor-Inare III </a><br />
344  *  ☭ Hial Atropa!! ☭
345  * @version 20130118
346  * @param {Array} arr The array to calculate frequencies from.
347  * @returns {Object} Returns an object whose keys are each unique
348  *  elements from the array and their value is their frequency of
349  *  occurrence within the array. Be careful that your array does
350  *  not contain values matching object instance property names.
351  * @example
352  * var x = [1,1,1,1,1,3,3];
353  * atropa.arrays.getFrequency(x);
354  * // returns {
355  * //     "1": 5,
356  * //     "3": 2
357  * // }
358  * @example
359  * var x = ["bill", "fred", "fred", "jane"];
360  * atropa.arrays.getFrequency(x);
361  * // returns {
362  * //     "bill": 1,
363  * //     "fred": 2,
364  * //     "jane": 1
365  * // }
366  * @example
367  * var x = [1,3,{'aProp' : 'aVal'}];
368  * atropa.arrays.getFrequency(x);
369  * // returns {
370  * //     "1": 1,
371  * //     "3": 1,
372  * //     "[object Object]": 1
373  * // }
374  * @example
375  * var obj = {'aProp' : 'aVal'};
376  * var otherObj = {};
377  * var x = [1,3,obj,otherObj,{'aDoughnut' : 'sprinkles'}];
378  * atropa.arrays.getFrequency(x);
379  * // returns {
380  * //     "1": 1,
381  * //     "3": 1,
382  * //     "[object Object]": 3
383  * // }
384  * @example
385  * var x = [1,3,"toString"];
386  * atropa.arrays.getFrequency(x);
387  * // returns {
388  * //     "1": 1,
389  * //     "3": 1,
390  * //     "toString": "function toString() {\n    [native code]\n}1"
391  * // }
392  */
393 atropa.arrays.getFrequency = function (arr) {
394     "use strict";
395     var out = arr.reduce(function (acc, curr) {
396         if (acc[curr] === undefined) {
397             acc[curr] = 1;
398         } else {
399             acc[curr] += 1;
400         }
401         return acc;
402     }, {});
403     return out;
404 };

463 /**
464  * Reindexes an array.
465  * @author <a href="mailto:matthewkastor@gmail.com">
466  *  Matthew Christopher Kastor-Inare III </a><br />
467  *  ☭ Hial Atropa!! ☭
468  * @version 20130118
469  * @param {Array} arr The array with discontinuous keys.
470  * @returns {Array} Returns an array with continuous keys.
471  * @example
472  * var x = [ "a", "b", "c", undefined ];
473  * console.log(x); // [ "a", "b", "c", undefined ]
474  * console.log(x.length); // 4
475  * 
476  * delete x[1]; // deletes the key from the array but
477  *              // the array length remains the same
478  *              // at this point the arrays keys are 0, 2, and 3
479  * console.log(x); // [ "a", undefined, "c", undefined ]
480  * console.log(x.length); // 4
481  * 
482  * x = atropa.arrays.reindex(x);
483  * console.log(x); //  [ "a", "c", undefined ]
484  *    // note that the last element existed in the array, its value was
485  *    // undefined but it did have a key so the element remains in the array.
486  *    //
487  *    // The deleted element was in fact deleted from the array so there was no
488  *    // key x[1] at all, when trying to access this non existing element the
489  *    // value of undefined was returned. This behavior is confusing unless you
490  *    // think about the arrayas an object whose properties are named by
491  *    // numbers. Accessing an undefined property returns undefined regardless
492  *    // of whether the property existed in the past or not.
493  * console.log(x.length); // 3
494  */
495 atropa.arrays.reindex = function reindex(arr) {
496     "use strict";
497     var idx, out;
498     out = [];
499     for(idx in arr) {
500         if(arr.hasOwnProperty(idx)) {
501             out.push(arr[idx]);
502         }
503     }
504     return out;
505 };
506 /**

1208     if(true === outputHTML) {
1209         target = target.replace(
1210             /(\. ?){2,}/gi,
1211             '<span style="color : brown ;"> [shit taco] </span>'
1212         );
1213         target = '<p> ' + target.replace(/(\r\n|\r|\n)/g,' <br/> ') + ' </p>';
1214     } else {
1215         target = target.replace(/(\. ?){2,}/gi, ' [shit taco] ');
1216     }
1217     /**
1218      * Accepts plain text input and Gloriously WTFifies it.
1219      * @author <a href="mailto:matthewkastor@gmail.com">
1220      *  Matthew Christopher Kastor-Inare III </a><br />
1221      *  ☭ Hial Atropa!! ☭
1222      * @version 20130112
1223      * @methodOf atropa.wtf.wtfify-
1224      * @private
1225      * @param {String} m First matched pattern in string searched.
1226      * @param {String} sub1 First matched subpattern in string searched.
1227      * @param {String} sub2 Second matched subpattern in string searched.
1228      */
1229     replacementText = function (m, sub1, sub2) {
1230         wtfCount++;
1231         sub1 = atropa.setAsOptionalArg('', sub1);
1232         sub2 = atropa.setAsOptionalArg('', sub2);
1233         var out;
1234         if(true === outputHTML) {
1235             out = '<span style="color : red ;">' +
1236                 sub1 + atropa.wtf.dictionary[word] + sub2 +
1237                 '</span>';
1238         } else {
1239             out = sub1 + atropa.wtf.dictionary[word] + sub2;
1240         }
1241         return out;
1242     };
1243     for (word in atropa.wtf.dictionary) {
1244         if (atropa.wtf.dictionary.hasOwnProperty(word)) {
1245             oldWord = atropa.regex.appendPrefixesAndSuffixes(word);
1246             regexValue = new RegExp(oldWord, 'gi');
1247             target = target.replace(regexValue, replacementText);
1248         }
1249     }
1250     ret.wtfCount = wtfCount;
1251     ret.wordCount = wordCount;
1252     ret.score = wtfCount / wordCount;
1253     ret.txt = target;
1254     return ret;
1255 };

1482 /**
1483  * Appends common prefix, suffix, and word boundary regex strings to
1484  * the supplied word.
1485  * @author <a href="mailto:matthewkastor@gmail.com">
1486  *  Matthew Christopher Kastor-Inare III </a><br />
1487  *  ☭ Hial Atropa!! ☭
1488  * @version 20130110
1489  * @param {String} word The word to append prefix and suffix to
1490  * @param {Integer} threshold The word.length at which it does not
1491  * make sense to append prefix and suffix. Defaults to 3.
1492  * @returns {String} Returns the supplied word with prefix, suffix,
1493  * and word boundaries attached. If the word.length was not greater
1494  * than the threshold, only word boundaries are attached. The string
1495  * represents a RegEx which should pick out most forms of regular
1496  * words.
1497  */
1498 atropa.regex.appendPrefixesAndSuffixes = function (word, threshold) {
1499     "use strict";
1500     var prefixes,
1501     suffixes;
1502     prefixes = '(pre|un|re)?';
1503     suffixes = '(ification|' +
1504                 'tionally|' +
1505                 'ication|' +
1506                 'ified|istic|iness|' +
1507                 'fare|tion|ance|ence|less|ally|able|ness|ized|ised|' +
1508                 'ous|ify|ing|ity|ful|ant|ate|est|ism|izm|ist|' +
1509                 'ic|al|ed|er|et|ly|rs|in|' +
1510                 'y|s|r|d)?';
1511     
1512     threshold = threshold === undefined ? 3 : threshold;
1513     
1514     if (word.length > threshold) {
1515         word = '\\b' + prefixes + word + suffixes + '\\b';
1516     } else {
1517         word = '\\b()' + word + '()\\b';
1518     }
1519     return word;
1520 };
