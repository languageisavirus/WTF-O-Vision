 37 /**
 38  * Appends common prefix, suffix, and word boundary regex strings to
 39  * the supplied word.
 40  * @author <a href="mailto:matthewkastor@gmail.com">
 41  *  Matthew Christopher Kastor-Inare III </a><br />
 42  *  ☭ Hial Atropa!! ☭
 43  * @version 20130110
 44  * @param {String} word The word to append prefix and suffix to
 45  * @param {Integer} threshold The word.length at which it does not
 46  * make sense to append prefix and suffix. Defaults to 3.
 47  * @returns {String} Returns the supplied word with prefix, suffix,
 48  * and word boundaries attached. If the word.length was not greater
 49  * than the threshold, only word boundaries are attached. The string
 50  * represents a RegEx which should pick out most forms of regular
 51  * words.
 52  */
 53 atropa.regex.appendPrefixesAndSuffixes = function (word, threshold) {
 54 	"use strict";
 55 	var prefixes,
 56 	suffixes;
 57 	prefixes = '(pre|un|re)?';
 58 	suffixes = '(ification|tionally|ication|ified|istic|iness|fare|tion|ance|ence|less|ally|able|ness|ized|ised|ous|ify|ing|ity|ful|ant|ate|est|ism|izm|ist|ic|al|ed|er|et|ly|rs|in|y|s|r|d)?';
 59 	
 60 	threshold = threshold === undefined ? 3 : threshold;
 61 	
 62 	if (word.length > threshold) {
 63 		word = '\\b' + prefixes + word + suffixes + '\\b';
 64 	} else {
 65 		word = '\\b()' + word + '()\\b';
 66 	}
 67 	return word;
 68 };

 79  *   function(optionalArg) {

 81  *       optionalArg = atropa.setAsOptionalArg(defaultVal, optionalArg);
 82  *       return optionalArg;

 84  * </pre>
 85  * @author <a href="mailto:matthewkastor@gmail.com">
 86  *  Matthew Christopher Kastor-Inare III </a><br />
 87  *  ☭ Hial Atropa!! ☭
 88  * @version 20120909
 89  * @param {Mixed} defaultVal The default value to set.
 90  * @param {Mixed} optionalArg A reference to the optional argument.
 91  * @returns {Mixed} Returns the default value supplied when the optional
 92  * argument is undefined or null. Otherwise, the supplied optional argument
 93  * is returned.
 94  */
 95 atropa.setAsOptionalArg = function (defaultVal, optionalArg) {
 96 	"use strict";
 97 	if (optionalArg === undefined || optionalArg === null) {
 98 		optionalArg = defaultVal;
 99 	}
100 	return optionalArg;
101 };

170  *  ☭ Hial Atropa!! ☭

172  * @param {Number} offset The amount of spaces you want removed 
173  * from the beginning of the text.
174  * @returns Returns the processed text.

176 atropa.string.offsetWhiteSpace = function offsetWhiteSpace(text, offset) {
177 	'use strict';
178     var regx;
179     regx = new RegExp('^ {' + offset + '}');
180     text = text.replace(regx, '');
181     return text;

184 /**
185  * Converts all tabs in leading whitespace into four spaces.
186  * @author <a href="mailto:matthewkastor@gmail.com">
187  *  Matthew Christopher Kastor-Inare III </a><br />
188  *  ☭ Hial Atropa!! ☭
189  * @param {String} text The text to process
190  * @returns {String} Returns the processed text.
191  */
192 atropa.string.normalizeWhiteSpacePrefix = function normalizeWhiteSpacePrefix(text) {
193 	'use strict';
194     var prefix = text.match(/^\s*/);
195     if(prefix) {
196         prefix = prefix[0];
197         prefix = prefix.replace(/\t/g, '    ');
198         text = text.replace(/^\s*/, prefix);
199     }
200     return text;
201 };

217 /**
218  * Counts the number of leading space or tab characters but not both.
219  * @author <a href="mailto:matthewkastor@gmail.com">
220  *  Matthew Christopher Kastor-Inare III </a><br />
221  *  ☭ Hial Atropa!! ☭
222  * @param {String} text The text to analyze.
223  * @returns {Number} Returns the quantity of leading spaces or tabs.
224  */
225 atropa.string.getOffset = function getOffset(text) {
226 	'use strict';
227     var offset = 0,
228         leadingChar = text.charAt(0);
229         
230     if( leadingChar === ' ' || leadingChar === '\t') {
231         while(text.charAt(offset) === leadingChar && offset < text.length) {
232             offset++;
233         }
234     }
235     return offset;
236 };
237 /**
238  * Breaks a string into an array of words.
239  * @author <a href="mailto:matthewkastor@gmail.com">
240  *  Matthew Christopher Kastor-Inare III </a><br />
241  *  ☭ Hial Atropa!! ☭
242  * @version 20130118
243  * @param {String} text The text to analyze.
244  * @returns {Array} Returns an array of the words in
245  *  the given text.
246  */
247 atropa.string.getWords = function (text) {
248 	"use strict";
249 	var out = [];
250 	function invalidChars(element) {
251 		var matched = /^[\-'’`]+$/.test(element);
252 		// invert the result of test. throw out elements that match.
253 		return !matched;
254 	}
255 	out = atropa.arrays.removeEmptyElements(
256 		text.split(/[^A-Za-z\-'’`]+/gi)
257 	);
258 	out = out.filter(invalidChars);
259 	return out;
260 };
261 /**
262  * Escapes <code>CDATA</code> sections in text
263  *  so that the text may be embedded into a 
264  *  <code>CDATA</code> section. This should be run
265  *  on any text which may contain the string 
266  *  <code>]]></code> since said string will effectively
267  *  end the <code>CDATA</code> section prematurely.
268  * @author <a href="mailto:matthewkastor@gmail.com">
269  *  Matthew Christopher Kastor-Inare III </a><br />
270  *  ☭ Hial Atropa!! ☭
271  * @version 20130118
272  * @param {String} text The text containing 
273  *  <code>CDATA</code> sections to escape.
274  * @returns {Array} Returns a string with escaped
275  *  <code>CDATA</code> sections.
276  * @see <a href="http://en.wikipedia.org/wiki/CDATA#Nesting">
277  *  http://en.wikipedia.org/wiki/CDATA#Nesting</a>
278  * @see <a href="https://bugzilla.mozilla.org/show_bug.cgi?id=98168">
279  *  https://bugzilla.mozilla.org/show_bug.cgi?id=98168</a>
280  */
281 atropa.string.escapeCdata = function escapeCdata(text) {
282 	return String(text).replace(/]]>/g, ']]]]><![CDATA[>');
283 };

765 /**
766  * Accepts plain text input and Gloriously WTFifies it.
767  * @author <a href="mailto:matthewkastor@gmail.com">
768  *  Matthew Christopher Kastor-Inare III </a><br />
769  *  ☭ Hial Atropa!! ☭
770  * @version 20130110
771  * @param {String} target The text to WTFify.
772  * @return {String} Returns Genuine WTFified text.
773  */
774 atropa.wtf.wtfify = function (target, isHTML) {
775 	"use strict";
776 	var regexValue,
777 	replacementText,
778 	x,
779 	oldWord,
780 	wtfCount,
781 	wordCount,
782 	ret;
783 	
784 	if(true !== isHTML) {
785 		isHTML = false;
786 	}
787 	ret = {};
788 	wtfCount = 0;
789 	target = target.trim();
790 	target = target.replace(/(\. ?){2,}/gi, '<span style="color : brown ;"> [shit taco] </span>');
791 	target = target.replace(/\b[ivxcl]+\./gi, '<span style="color : brown ;"> [#!~ syntax error : unexpected shit taco ~!#] </span>');
792 	if(true === isHTML) {
793 		target = '<p> ' + target.replace(/(\r\n|\r|\n)/g,' <br/> ') + ' </p>';
794 	}
795 	wordCount = atropa.string.countWords(target);
796 	/**
797 	 * Accepts plain text input and Gloriously WTFifies it.
798 	 * @author <a href="mailto:matthewkastor@gmail.com">
799 	 *  Matthew Christopher Kastor-Inare III </a><br />
800 	 *  ☭ Hial Atropa!! ☭
801 	 * @version 20130112
802 	 * @methodOf atropa.wtf.wtfify-
803 	 * @private
804 	 * @param {String} m First matched pattern in string searched.
805 	 * @param {String} sub1 First matched subpattern in string searched.
806 	 * @param {String} sub2 Second matched subpattern in string searched.
807 	 */
808 	replacementText = function (m, sub1, sub2) {
809 		wtfCount++;
810 		sub1 = atropa.setAsOptionalArg('', sub1);
811 		sub2 = atropa.setAsOptionalArg('', sub2);
812 		var out;
813 		if(true === isHTML) {
814 			out = '<span style="color : red ;">' + sub1 + atropa.wtf.dictionary[x] + sub2 + '</span>';
815 		} else {
816 			out = sub1 + atropa.wtf.dictionary[x] + sub2;
817 		}
818 		return out;
819 	};
820 	for (x in atropa.wtf.dictionary) {
821 		if (atropa.wtf.dictionary.hasOwnProperty(x)) {
822 			oldWord = atropa.regex.appendPrefixesAndSuffixes(x);
823 			regexValue = new RegExp(oldWord, 'gi');
824 			target = target.replace(regexValue, replacementText);
825 		}
826 	}
827 	ret.wtfCount = wtfCount;
828 	ret.wordCount = wordCount;
829 	ret.score = wtfCount / wordCount;
830 	ret.txt = target;
831 	return ret;
832 };
833 /**
834  * WTFifies the <code>textContent</code> or <code>value</code> of the
835  *  given element and replaces the element with a pre block
836  *  containing the results of WTFification.
837  * @param {HTMLElement} elementReference A reference to an HTML Element.
838  * @version 20130112
839  */
840 atropa.wtf.htmlElement = function (elementReference) {
841 	"use strict";
842 	var wtfified, txt;
843 	elementReference.innerHTML = elementReference.innerHTML.replace(/<br>(\s+)?\r?\n?/g, '\r\n');
844 	txt = elementReference.value || elementReference.textContent;
845 	wtfified = atropa.wtf.wtfify(txt, true);
846 	elementReference.innerHTML = '<pre style="color:black; background:white; white-space:pre-wrap;">' + wtfified.txt + '</pre>';
847 };
